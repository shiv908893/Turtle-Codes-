import turtle
ninja=turtle.Turtle()
ninja.speed(25)

for i in range(100):
    ninja.forward(100)
    ninja.right(30)
    ninja.forward(60)
    ninja.left(60)
    ninja.forward(100)
    ninja.right(30)

    ninja.penup()
    ninja.setposition(0, 0)
    ninja.pendown()

    ninja.right(10)
turtle.done()
